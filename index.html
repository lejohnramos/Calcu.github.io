from pyscript import Element def calculate_final_grade(prelim, total_unpredictable): total = prelim * 0.2 intermediate_total = total_unpredictable - total final_grade = intermediate_total / 0.8 return final_grade def show_result(event): prelim_value = float(Element("prelim").value) total_unpredictable_value = float(Element("total_unpredictable").value) final_grade = calculate_final_grade(prelim_value, total_unpredictable_value) Element("result").element.innerHTML = f"Final grade required: {final_grade:.2f}" # Attach the event listener to the button def setup(): Element("calculate").element.onclick = show_result setup() # Call the setup function
