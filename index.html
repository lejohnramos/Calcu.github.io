from pyscript import Element def calculate_final_grade(prelim, total_unpredictable): total = prelim * 0.2 intermediate_total = total_unpredictable - total final_grade = intermediate_total / 0.8 return final_grade def show_result(event): prelim_element = Element("prelim") total_unpredictable_element = Element("total_unpredictable") result_element = Element("result") try: prelim = float(prelim_element.value) total_unpredictable = float(total_unpredictable_element.value) final_grade = calculate_final_grade(prelim, total_unpredictable) result_element.element.innerHTML = f"Final grade required: {final_grade}" except ValueError: result_element.element.innerHTML = "Invalid input, please enter valid numbers." # Add event listener Element("calculate").element.onclick = show_result
