def calculate_final_grade(prelim, total_unpredictable): total = prelim * 0.2 intermediate_total = total_unpredictable - total final_grade = intermediate_total / 0.8 return final_grade def show_result(event): prelim = float(document.getElementById("prelim").value) total_unpredictable = float(document.getElementById("total_unpredictable").value) final_grade = calculate_final_grade(prelim, total_unpredictable) document.getElementById("result").innerHTML = f"Final grade required: {final_grade}" document.getElementById("calculate").addEventListener("click", show_result)
